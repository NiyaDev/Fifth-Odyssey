namespace = fo_bl_goals


fo_bl_goals.1 = { # Noble bonuses
  type = country_event
  title = fo_bl_goals.1.title
  desc = fo_bl_goals.1.desc

  trigger = {
    situation:fo_fifth_odyssey.var:nobles_count >= 250
  }

  dynamic_historical_event = {
  	tag = EXO
 		from = 1337.1.1
 		to = 1350.1.1
 		monthly_chance = 100
 	}
 	fire_only_once = yes

  option = {
    name = fo_bl_goals.1.a
    custom_tooltip = { text = fo_bl_goals.1.a }
    add_country_modifier = {
      mode = add_and_replace
      modifier = fo_goals_noblebonuses
      years = -1
    }
  }
}
