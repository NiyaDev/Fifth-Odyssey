fo_noble_flight.10 = {
  type = country_event
  category = situation_event
  title = fo_noble_flight.10.title
  desc = fo_noble_flight.10.desc

  # Combat ships
  option = {
    name = fo_noble_flight.10.a
    trigger = { situation:fo_noble_flight.var:wealth >= 40 }

    situation:fo_noble_flight = {
  		set_variable = {
				name = wealth
				value = {
					add = var:wealth
					add = -40
				}
			}
			set_variable = {
			  name = combat
			  value = {
			    add = var:combat
			    add = 1
			  }
			}
		}
		custom_tooltip = { text = fo_noble_flight.10.aa }
  }

  # Resource ships
  option = {
    name = fo_noble_flight.10.b
    trigger = { situation:fo_noble_flight.var:wealth >= 50 }

    situation:fo_noble_flight = {
  		set_variable = {
				name = wealth
				value = {
					add = var:wealth
					add = -50
				}
			}
			set_variable = {
			  name = resource
			  value = {
			    add = var:resource
			    add = 1
			  }
			}
		}
		custom_tooltip = { text = fo_noble_flight.10.bb }
  }

  # Pop ship
  option = {
    name = fo_noble_flight.10.c
    trigger = { situation:fo_noble_flight.var:wealth >= 30 }

    situation:fo_noble_flight = {
  		set_variable = {
				name = wealth
				value = {
					add = var:wealth
					add = -30
				}
			}
			set_variable = {
			  name = pop
			  value = {
			    add = var:pop
			    add = 1
			  }
			}
		}
		custom_tooltip = { text = fo_noble_flight.10.cc }
  }

  # Cancel
  option = {
    name = fo_noble_flight.10.d
    custom_tooltip = { text = fo_noble_flight.10.dd }
  }
}
