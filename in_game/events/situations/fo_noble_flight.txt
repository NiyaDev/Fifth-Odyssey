      limit = { dysaster_type:byzantine_succession_crisis = { disaster_is_active = yes } c:BYZ = { num_loans >= 4 } }
      random_list = {
        50 = {
          situation:fo_noble_flight = { set_variable = { name = wealth value = { add = var:wealth add = -60 } } }
          trigger_event_non_silently = { id = fo_noble_flight.15 }
        }

    }
    else_if = {
      limit = { dysaster_type:byzantine_succession_crisis = { disaster_is_active = yes } c:BYZ = { num_loans >= 2 } }
      random_list = {
        25 = {
          situation:fo_noble_flight = { set_variable = { name = wealth value = { add = var:wealth add = -30 } } }
          trigger_event_non_silently = { id = fo_noble_flight.15 }

      random_list = {
        5  = {
          situation:fo_noble_flight = { set_variable = { name = wealth value = { add = var:wealth add = -30 } } }

fo_noble_flight.19 = { # Stealing Byzantine Silk 3
  type = country_event
  title = fo_noble_flight.19.title
  desc = fo_noble_flight.19.desc
  
  option = {
    name = fo_noble_flight.19.a

    if = { # You didin't even try
      limit = { situation:fo_noble_flight.var:heist = 0 }
      custom_tooltip = { text = fo_noble_flight.19.aa }
      situation:fo_noble_flight = { set_variable = { name = heist_success  value = no } }
      hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 } }
    }
    else_if = { # Ver low
      limit = { situation:fo_noble_flight.var:heist < 2 }
      custom_tooltip = { text = fo_noble_flight.19.ab }
      hidden_effect = {
        random_list = {
          10 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = yes } } }
          90 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = no } } }
        }
        hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 days = 30 } }
      }
    }
    else_if = { # low
      limit = { situation:fo_noble_flight.var:heist < 4 }
      custom_tooltip = { text = fo_noble_flight.19.ac }
      hidden_effect = {
        random_list = {
          30 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = yes } } }
          70 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = no } } }
        }
        hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 days = 30 } }
      }
    }
    else_if = { # mediocre
      limit = { situation:fo_noble_flight.var:heist < 6 }
      custom_tooltip = { text = fo_noble_flight.19.ad }
      hidden_effect = {
        random_list = {
          50 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = yes } } }
          50 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = no } } }
        }
        hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 days = 30 } }
      }
    }
    else_if = { # good
      limit = { situation:fo_noble_flight.var:heist < 8 }
      custom_tooltip = { text = fo_noble_flight.19.ae }
      hidden_effect = {
        random_list = {
          70 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = yes } } }
          30 = { situation:fo_noble_flight = { set_variable = { name = heist_success  value = no } } }
        }
        hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 days = 30 } }
      }
    }
    else = { # Perfect
      custom_tooltip = { text = fo_noble_flight.19.af }
      hidden_effect = {
        situation:fo_noble_flight = { set_variable = { name = heist_success  value = yes } }
        hidden_effect = { trigger_event_non_silently = { id = fo_noble_flight.20 days = 30 } }
      }
    }
  }
}
