# Random Events
fo_fifth_odyssey.20 = { # Nobles join
  type = country_event
  title = fo_fifth_odyssey.20.title
  desc = fo_fifth_odyssey.20.desc
 	category = situation_event

  option = { # Accept them no conditions
    name = fo_fifth_odyssey.20.a
    custom_tooltip = { text = fo_fifth_odyssey.20.aa }

    situation:fo_fifth_odyssey = {
      set_variable = {
        name = nobles_additional
        value = {
          add = var:nobles_additional
          add = 20
        }
      }
    }
  }
  option = { # Accept them for payment
    name = fo_fifth_odyssey.20.b
    custom_tooltip = { text = fo_fifth_odyssey.20.bb }

    add_gold = 100

    situation:fo_fifth_odyssey = {
      set_variable = {
        name = nobles_additional
        value = {
          add = var:nobles_additional
          add = 10
        }
      }
    }
  }
  option = { # Accept them for money but lie
    name = fo_fifth_odyssey.20.c
    custom_tooltip = { text = fo_fifth_odyssey.20.cc }

    add_gold = 100
  }
  option = { # Deny them
    name = fo_fifth_odyssey.20.d
    custom_tooltip = { text = fo_fifth_odyssey.20.cc }
  }
}
