fo_fifth_odyssey.26 = { # Peasants flee Frankokratic country
  type = country_event
  title = fo_fifth_odyssey.26.title
  desc = fo_fifth_odyssey.26.desc
  category = situation_event
 	outcome = positive

  immediate = {
    random_country = {
      limit = {
        culture != culture:greek_culture
        culture != culture:turkish_culture
        religion != religion:orthodox
        dominant_culture = culture:greek_culture
      }
      save_scope_as = target_country
    }
  }

  option = { # Invite them
    name = fo_fifth_odyssey.26.a
    custom_tooltip = { text = fo_fifth_odyssey.26.aa }

    scope:target_country = {
      add_opinion = {
        modifier = opinion_invited_peasants_soft
        target = c:EXO
      }
    }
    
    situation:fo_fifth_odyssey = {
      set_variable = {
        name = peasants_additional
        value = {
          add = var:peasants_additional
          add = 100
        }
      }
    }
  }
  option = { # Refuse them
    name = fo_fifth_odyssey.26.b
    custom_tooltip = { text = fo_fifth_odyssey.26.bb }

    scope:target_country = {
      add_opinion = {
        modifier = opinion_refused_peasants
        target = c:EXO
      }
    }
  }
}

fo_fifth_odyssey.40 = { # Romanyoti approach
  type = country_event
  title = fo_fifth_odyssey.40.title
  desc = fo_fifth_odyssey.40.desc
  fire_only_once = yes
 	outcome = positive

  dynamic_historical_event = {
    tag = EXO
    from = 1337.1.1
    to = 1350.1.1
    monthly_chance = 5
  }
  
  trigger = {
    has_or_had_tag = EXO
    situation:fo_fifth_odyssey = { situation_is_active = yes }
    situation:fo_fifth_odyssey.var:romanyoti_joined = no
    situation:fo_fifth_odyssey.var:clergy_count <= 0
  }

  option = { # Allow
    name = fo_fifth_odyssey.40.a
    custom_tooltip = { text = fo_fifth_odyssey.40.aa }

    add_gold = 50
    add_tolerated_culture = culture:romanyoti

    situation:fo_fifth_odyssey = {
      set_variable = {
        name = romanyoti_joined
        value = yes
      }
      set_variable = {
        name = romanyoti_count
        value = 200
      }
    }
  }
  option = { # Refuse
    name = fo_fifth_odyssey.40.b
    custom_tooltip = { text = fo_fifth_odyssey.40.bb }
  }
}
