namespace = fo_noble_flight

fo_noble_flight_funding = {
  type = situation
  show_message = no
  
  allow = {
    scope:actor = {
      "estate(estate_type:nobles_estate)" = {
      	estate_gold > {
      		value = 20
      		multiply = 1
      	}
      }
    }

  }

	cooldown = {
		type = funding_cooldown_key
		months = 4
	}

  select_trigger = {
		looking_for_a = situation
		interaction_source_list = { situation:fo_noble_flight = { add_to_list = source } }
		target_flag = recipient
		name = "choose_situation"
		column = { data = name }
		visible = {
			situation:fo_noble_flight = this
			situation_is_active = yes
		}
	}

  effect = {
  	situation:fo_noble_flight = {
  		set_variable = {
				name = fo_noble_flight_wealth
				value = {
					add = var:fo_noble_flight_wealth
					add = 20
				}
			}
		}
    scope:actor = {
      "estate(estate_type:nobles_estate)" = {
		  	estate_add_gold = {
		  		value = {
		  			add = -20
		  			multiply = 1
		  		}
		  	}
		  }
		  estate_satisfaction:nobles_estate = {
		  	subtract = 0.1
		  }
		}
  }
}
