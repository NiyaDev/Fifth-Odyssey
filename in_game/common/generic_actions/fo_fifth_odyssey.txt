#namespace = fo_fifthodyssey_actions

fo_fifthodyssey_buy_ships = {
  type = situation
  show_message = no

  select_trigger = {
    looking_for_a = situation
    interaction_source_list = { situation:fo_fifth_odyssey = { add_to_list = source } }
    target_flag = recipient
    name = "choose_situation"
    column = { data = name }
    visible = {
      situation:fo_fifth_odyssey = this
      situation_is_active = yes
    }
  }
  allow = {
    c:EXO.gold > 20
  }
	cooldown = {
		type = funding_cooldown_key
		months = 4
	}

  effect = {
    scope:actor = { add_gold = -20 }
    situation:fo_fifth_odyssey = {
      set_variable = {
        name = ships
        value = {
          add = situation:fo_fifth_odyssey.var:ships
          add = 1
        }
      }
    }
  }
}

fo_fifthodyssey_leave = {
  type = situation
  show_message = no

  select_trigger = {
    looking_for_a = situation
    interaction_source_list = { situation:fo_fifth_odyssey = { add_to_list = source } }
    target_flag = recipient
    name = "choose_situation"
    column = { data = name }
    visible = {
      situation:fo_fifth_odyssey = this
      situation_is_active = yes
    }
  }
  allow = {
    situation:fo_fifth_odyssey = {
      or = {
        not = { has_variable = started_voyage }
        var:started_voyage = no
      }
    }
  }

  effect = {
    scope:actor = { trigger_event_non_silently = fo_fifth_odyssey.1000 }
    situation:fo_fifth_odyssey = {
      set_variable = {
        name = started_voyage
        value = yes
      }
    }
  }
}
