
fo_noble_flight_update_province = {
  location:patawomeck = {
    change_location_owner = c:FO0
    hidden_effect = { add_core = c:FO0 }

    # Development
    change_development = {
      add = {
        value = {
          add = situation:fo_noble_flight.var:resource
          multiply = 0.1
        }
      }
    }

    # Population
    add_pop = { # Peasants
      culture = culture:greek_culture
      religion = religion:orthodox
      type = pop_type:peasants
      size = {
        value = situation:fo_noble_flight.var:pop
        multiply = 0.49
      }
    }
    add_pop = { # Nobles
      culture = culture:greek_culture
      religion = religion:orthodox
      type = pop_type:nobles
      size = {
        value = situation:fo_noble_flight.var:pop
        multiply = 0.005
      }
    }

    # Change name
    if = {
      limit = { situation:fo_noble_flight.var:current_leader = character:fo_romaboo }
      rename_location = new_rome_gr
    }
    if = {
      limit = { situation:fo_noble_flight.var:current_leader = character:fo_byzaboo }
      rename_location = new_constantinople_gr
    }
    if = {
      limit = { situation:fo_noble_flight.var:current_leader = character:fo_alexaboo }
      rename_location = new_pella_gr
    }
  }
}
