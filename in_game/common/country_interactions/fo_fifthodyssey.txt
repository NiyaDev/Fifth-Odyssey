
ask_for_knowledge = {
  type = diplomacy
  category = CATEGORY_ECONOMY_ACTIONS
  diplomatic_cost = ask_for_knowledge_cost
  ai_tick = never

  potential = { situation:fo_fifth_odyssey.var:can_request_knowledge ?= yes }
  cooldown = {
    type = ask_for_knowledge
    years = 5
  }

  allow = {
    scope:actor = {
      gold >= 60
    }
  }
  select_trigger = {
    looking_for_a = country
    target_flag = recipient
    name = "choose_country"
    column = { data = name }
    visible = { has_or_had_tag = VEN }
    enabled = {
      not = { is_at_war_with = scope:actor }
      opinion = {
        target = scope:actor
        value > 120
      }
    }
  }

  effect = {
    scope:actor = {
      add_gold = -60
      trigger_event_non_silently = fo_fifth_odyssey.101
    }
  }
}

steal_knowledge = {
	type = diplomacy
	category = CATEGORY_COVERT_ACTIONS
	ai_tick = never

  potential = { situation:fo_fifth_odyssey.var:can_request_knowledge ?= yes }
  cooldown = {
    type = steal_knowledge
    years = 5
  }

  allow = {
    scope:actor = {
      spy_network = {
        target = c:VEN
        value >= 50
      }
    }
  }

  select_trigger = {
    looking_for_a = country
    target_flag = recipient
    name = "choose_country"
    column = { data = name }
    visible = { has_or_had_tag = VEN }
    enabled = {
      not = { is_at_war_with = scope:actor }
      spy_network = {
        target = scope:actor
        value >= 50
      }
    }
  }

  effect = {
    scope:actor = {
      add_spy_network = {
        target = c:VEN
        value = -50
      }
      random_list = {
        50 = { trigger_event_non_silently = fo_fifth_odyssey.102 }
        50 = { trigger_event_non_silently = fo_fifth_odyssey.103 }
      }
    }
  }
}
