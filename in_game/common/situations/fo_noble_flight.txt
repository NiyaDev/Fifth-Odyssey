fo_noble_flight = {
  monthly_spawn_chance = 0

  can_end = {
    current_year > 1339
  }

  visible = {
    exists = c:BYZ
    tag = BYZ
  }

  on_starting = {
    set_variable = {
      name = religion
      value = religion:orthodox
    }
  }
  on_ending = {
    c:BYZ = { trigger_event_non_silently = fo_noble_flight.100 }
  }

  on_monthly = {
    situation:fo_noble_flight = {
      # Resource ships
      set_variable = {
        name = dev_output
        value = {
          add = var:resource
          multiply = 0.1
        }
      }
      set_variable = {
        name = dev_money
        value = {
          add = var:resource
          multiply = 25
        }
      }

      # Population ships
      set_variable = {
        name = population_comm
        value = {
          add = var:pop
          multiply = 490
        }
      }
      set_variable = {
        name = population_nob
        value = {
          add = var:pop
          multiply = 10
        }
      }
    }
  }
}
